function doGet(e) { return processRequest(e); } function doPost(e) { return processRequest(e); } function processRequest(e) { try { var planilha = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Página1'); var params = {}; // Deteta se está a receber a foto (POST) ou apenas um alerta de desvio leve (GET) if (e.postData && e.postData.contents) { try { params = JSON.parse(e.postData.contents); } catch(x) { params = e.parameter; } } else { params = e.parameter; } var email = params.email || 'adriano.hennemann@gmail.com'; var nome = params.nome || 'Adriano'; var acao = params.acao || 'registro_inicial'; var foto = params.foto || ''; var dataHora = new Date(); var dados = planilha.getDataRange().getValues(); var linhaAluno = -1; // Procura o aluno na planilha for (var i = 1; i < dados.length; i++) { if (dados[i][2] && dados[i][2].toString().toLowerCase() === email.toLowerCase()) { linhaAluno = i + 1; break; } } var imagemFormula = 'SEM FOTO'; // Se a foto chegou, guarda no Drive e gera a fórmula para a Coluna A if (foto.length > 100) { try { var blob = Utilities.newBlob(Utilities.base64Decode(foto.split(',')[1]), 'image/jpeg', nome + '.jpg'); var arq = DriveApp.createFile(blob); arq.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); imagemFormula = '=IMAGE("https://drive.google.com/uc?id=' + arq.getId() + '")'; } catch(err) { imagemFormula = 'ERRO DRIVE'; } } if (linhaAluno === -1) { planilha.appendRow([imagemFormula, dataHora, email, nome, 'Render-ID', 0]); } else { if (acao === 'alerta_desvio') { var cel = planilha.getRange(linhaAluno, 6); cel.setValue((parseInt(cel.getValue()) || 0) + 1); } else if (acao === 'registro_inicial' && imagemFormula !== 'SEM FOTO') { planilha.getRange(linhaAluno, 1).setValue(imagemFormula); } planilha.getRange(linhaAluno, 2).setValue(dataHora); } return ContentService.createTextOutput("ok"); } catch(e) { return ContentService.createTextOutput("erro"); } }
