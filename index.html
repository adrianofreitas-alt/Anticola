const express = require('express'); const fs = require('fs'); const path = require('path'); const app = express(); app.use(express.json({ limit: '50mb' })); app.use(express.static('public')); app.post('/api/entrar', async (req, res) => { const { idAluno, fotoBase64, acao, email } = req.body; const acaoAtual = acao || 'registro_inicial'; const emailAluno = email || `${idAluno}@escola.com`; const nomeAluno = idAluno || 'Aluno'; if (!idAluno) return res.status(400).json({ erro: "Identificação do aluno ausente." }); if (fotoBase64) { try { // CORREÇÃO: Aceita tanto image/jpeg (do index.html) quanto image/png sem quebrar a string base64 const base64Data = fotoBase64.replace(/^data:image\/(png|jpeg);base64,/, ""); const nomeArquivo = `foto_${idAluno}_${Date.now()}.jpg`; fs.mkdirSync('./fotos', { recursive: true }); fs.writeFileSync(path.join('./fotos', nomeArquivo), base64Data, 'base64'); } catch (err) { console.error("Erro ao salvar foto local:", err); } } // URL do Apps Script obtida do navegador const urlAppsScript = "https://script.google.com/macros/s/AKfycbz_jK23n3C0BwE8_y6Lp_mE10Y-W9P-vP4N3z8M7h6Q/exec"; try { const response = await fetch(urlAppsScript, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ email: emailAluno, nome: nomeAluno, acao: acaoAtual, foto: fotoBase64 || '', nomeAba: 'Dados_Script' // <-- Mantido intacto conforme sua estrutura }) }); const texto = await response.text(); console.log("Resposta do Google Sheets:", texto); } catch (error) { console.error("Erro ao enviar para o Google Sheets:", error); } return res.json({ status: "sucesso", mensaje: "Dados processados!" }); }); const PORT = 3000; app.listen(PORT, () => { console.log(`Servidor rodando com sucesso em http://localhost:${PORT}`); });
