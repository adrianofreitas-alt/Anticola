// ATENÇÃO: Se quiser organizar, crie uma pasta no Drive e cole o ID aqui. // Deixe em branco "" para salvar na raiz. var FOLDER_ID = ""; function doGet(e) { return processRequest(e); } function doPost(e) { return processRequest(e); } function processRequest(e) { try { var planilha = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Página1'); if (!planilha) return ContentService.createTextOutput("Aba Página1 não encontrada."); // Deteta se os dados vieram via GET leve (desvio) ou POST pesado (foto) var params = (e.parameter && e.parameter.email) ? e.parameter : null; if (!params && e.postData) { // Tenta ler como JSON caso o navegador envie assim try { params = JSON.parse(e.postData.contents); } catch(x) {} } var email = params.email || 'adriano.hennemann@gmail.com'; // Fixo para teste var nome = params.nome || 'Adriano'; // Fixo para teste var acao = params.acao; var fotoBase64 = params.foto; // O texto da imagem comprimida var dataHora = new Date(); var dados = planilha.getDataRange().getValues(); var linhaAluno = -1; // Procura e-mail (Adriano) na Coluna C for (var i = 1; i < dados.length; i++) { if (dados[i][2] && dados[i][2].toString().toLowerCase() === email.toLowerCase()) { linhaAluno = i + 1; break; } } var imagemFormula = ""; // Se temos foto comprimida (vindo do POST), salvamos no Drive if (fotoBase64) { try { var imagemTexto = fotoBase64.split(',')[1] || fotoBase64; var blob = Utilities.newBlob(Utilities.base64Decode(imagemTexto), 'image/jpeg', email + '_' + dataHora.getTime() + '.jpg'); var pasta = (FOLDER_ID) ? DriveApp.getFolderById(FOLDER_ID) : DriveApp; var arquivo = pasta.createFile(blob); // Define permissões para qualquer pessoa com o link ver (necessário para =IMAGE() funcionar na planilha) arquivo.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); // Gera a fórmula para exibir a imagem na planilha imagemFormula = '=IMAGE("' + arquivo.getDownloadUrl() + '")'; } catch(err_foto) { imagemFormula = 'Erro na foto: ' + err_foto.toString(); } } // Estrutura baseada em image_0.png: // A: Foto (Formula IMAGE) | B: Data/Hora | C: Email | D: Nome | E: ID | F: Desvios if (linhaAluno === -1) { // Cria linha limpa planilha.appendRow([imagemFormula, dataHora, email, nome, 'Render-ID', 0]); } else { // Atualiza a foto se houver uma nova (no POST) if (imagemFormula) { planilha.getRange(linhaAluno, 1).setValue(imagemFormula); // Atualiza Coluna A } // Lida com desvios (no GET leve) na Coluna F if (acao === 'alerta_desvio') { var celulaDesvios = planilha.getRange(linhaAluno, 6); // Coluna F var valorAtual = parseInt(celulaDesvios.getValue()) || 0; celulaDesvios.setValue(valorAtual + 1); } // Sempre atualiza a data/hora na Coluna B planilha.getRange(linhaAluno, 2).setValue(dataHora); } return ContentService.createTextOutput("Sucesso").setMimeType(ContentService.MimeType.TEXT); } catch(err) { return ContentService.createTextOutput("Erro geral: " + err.toString()); } }
